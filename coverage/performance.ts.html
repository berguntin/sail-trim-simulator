/**
 * performance.ts — Cl/Cd → Qualitative Performance Estimate
 *
 * IMPORTANT: All numbers produced here are QUALITATIVE and ILLUSTRATIVE.
 * They are NOT predictions of real VMG, heel angle, or boat speed for any
 * real vessel. They exist solely to give the user intuitive feedback about
 * the direction and magnitude of trim changes. Do NOT use for navigation or
 * racing decisions.
 *
 * relativeVMG is normalised against a theoretical internal maximum obtained by
 * grid-searching the control space. It does NOT correspond to any polar.
 */
 
import type { AeroCoefficients, WindState, PerformanceEstimate } from './types'
import { computeSailShape } from './sailShape'
import { computeAeroCoefficients } from './aerodynamics'
 
// ---------------------------------------------------------------------------
// Theoretical maximum efficiency by wind speed (grid search)
// ---------------------------------------------------------------------------
 
/**
 * Grid-search over control combinations to find the peak efficiency
 * (Cl/Cd) achievable at a given wind state. The result is cached by wind
 * speed rounded to 1 kt because the search is relatively cheap but we
 * don't want to recompute it on every reactive tick.
 *
 * Grid resolution: 5 steps per control × 4 controls = 625 evaluations per
 * wind speed. Sufficient for a smooth relative scale; not a true optimisation.
 */
const _efficiencyCache = new Map<number, number>()
 
function maxTheoreticalEfficiency(wind: WindState): number {
  const key = Math.round(wind.trueWindSpeedKts)
  if (_efficiencyCache.has(key)) return _efficiencyCache.get(key)!
 
  const steps = [0, 25, 50, 75, 100]
  const travelerSteps = [-50, -25, 0, 25, 50]
  let maxEff = 0
 
  for (const mainsheet of steps) {
    for (const traveler of travelerSteps) {
      for (const cunningham of steps) {
        for (const backstay of steps) {
          const shape = computeSailShape(
            { mainsheet, traveler, cunningham, backstay },
            wind,
          )
          const { efficiency } = computeAeroCoefficients(shape, wind)
          if (efficiency > maxEff) maxEff = efficiency
        }
      }
    }
  }
 
  _efficiencyCache.set(key, maxEff)
  return maxEff
}
 
// ---------------------------------------------------------------------------
// relativeHeel
// ---------------------------------------------------------------------------
 
/**
 * Qualitative heel estimate: 0-100.
 *
 * Heeling force is proportional to aerodynamic lift (Cl) and to wind
 * pressure (∝ TWS²). We normalise against the theoretical maximum heel
 * force produceable (max Cl × max TWS²) to give a 0-100 scale.
 *
 * This is a gross simplification — real heel depends on righting moment,
 * displacement, waterline beam, etc. Here it is purely qualitative.
 */
function computeRelativeHeel(cl: number, wind: WindState): number {
  const { trueWindSpeedKts } = wind
  const heelForce = cl * (trueWindSpeedKts ** 2)
 
  // Max theoretical: Cl ≈ 1.5 at 25 kts
  const MAX_HEEL_FORCE = 1.5 * (25 ** 2) // = 937.5
  const heel = (heelForce / MAX_HEEL_FORCE) * 100
  return Math.max(0, Math.min(100, heel))
}
 
// ---------------------------------------------------------------------------
// powerBalance
// ---------------------------------------------------------------------------
 
/**
 * Simple three-state power classification.
 *
 * Based on relativeHeel thresholds:
 *  < 25  → underpowered  (not enough drive force)
 *  25-65 → optimal       (good balance of drive vs. heel)
 *  > 65  → overpowered   (too much heel, need to depower)
 *
 * Thresholds are deliberately qualitative. The breakpoints were chosen so
 * that "optimal" covers a realistic trim band and "overpowered" kicks in
 * where the heel force is clearly high.
 */
function classifyPowerBalance(
  relativeHeel: number,
): PerformanceEstimate['powerBalance'] {
  if (relativeHeel < 25) return 'underpowered'
  if (relativeHeel <= 65) return 'optimal'
  return 'overpowered'
}
 
// ---------------------------------------------------------------------------
// Public API
// ---------------------------------------------------------------------------
 
export function estimatePerformance(
  aero: AeroCoefficients,
  wind: WindState,
): PerformanceEstimate {
  const maxEff = maxTheoreticalEfficiency(wind)
  const relativeVMG = maxEff > 0
    ? Math.max(0, Math.min(100, (aero.efficiency / maxEff) * 100))
    : 0
 
  const relativeHeel = computeRelativeHeel(aero.cl, wind)
  const powerBalance = classifyPowerBalance(relativeHeel)
 
  return { relativeVMG, relativeHeel, powerBalance }
}
 
// Export internals for testing
export { maxTheoreticalEfficiency, computeRelativeHeel, classifyPowerBalance }
